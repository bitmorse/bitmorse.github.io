124  double seuil_d = p1->rayon + p2->rayon;

128 
129  if(d <= D_EQUILIBRE*seuil_d)
130  return calc_repulsion(d, seuil_d);
131  else if(d > D_EQUILIBRE*seuil_d && d <= D_FIN_ATTRACTION*seuil_d)
132  return calc_attraction(d, seuil_d);
133  else
134  return 0.;
135 }
